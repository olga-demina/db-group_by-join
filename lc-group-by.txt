1.1. Contare i corsi raggruppati per cfu

SELECT COUNT(*) AS `courses_number`, `courses`.`cfu` 
FROM `courses`
GROUP BY `courses`.`cfu`
ORDER BY `courses_number`;


1.2. Contare i corsi che hanno più di 10 cfu (raggrupati per cfu)

SELECT COUNT(*) AS `courses_number`, `courses`.`cfu` 
FROM `courses`
WHERE `courses`.`cfu` > 10
GROUP BY `courses`.`cfu`
ORDER BY `courses_number`;

1.3. Contare i corsi raggruppati per cfu e far vedere solo quelli cfu che hanno più di 100 corsi;

SELECT COUNT(*) AS `courses_number`, `courses`.`cfu` 
FROM `courses`
GROUP BY `courses`.`cfu`
HAVING `courses_number` > 100 
ORDER BY `courses_number`;

2. Contare gli studenti raggruppati per anno di nascita

SELECT COUNT(*) as `students_number`, YEAR(`students`.`date_of_birth`) as `birth_year`
FROM `students`
GROUP BY `birth_year`;

3.1 Selezionare la media dei voti per ogni appello d'esame

SELECT AVG(`exam_student`.`vote`) as `voto_medio`, `exam_student`.`exam_id`
FROM `exam_student`
GROUP BY `exam_student`.`exam_id`;


3.2 Selezionare il voto più basso dato ad ogni appello d'esame

SELECT MIN(`exam_student`.`vote`) as `voto_minimo`, `exam_student`.`exam_id`
FROM `exam_student`
GROUP BY `exam_student`.`exam_id`;


4. Contare gli appelli d'esame del mese di luglio raggruppati per giorno

SELECT COUNT(*) as `numero_appelli`, DAY(`exams`.`date`) as `giorno`
FROM `exams`
WHERE MONTH(`exams`.`date`) = 7
GROUP BY `giorno`;
